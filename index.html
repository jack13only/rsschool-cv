  function anagrams(word, words) {
    let sortedWord = word.split("").sort().join("");
    let objSorted = {};
    let result = [];
   
    for (let i = 0; i < words.length; i++) {
      let sorted = words[i].split("").sort().join("");
      objSorted[words[i]] = sorted;
    }
    
    for (let key in objSorted) {
      if (objSorted[key] == sortedWord) {
      result.push(key)
      }
    }
    return result;
  }
